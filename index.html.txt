<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Frog Jump Game</title>
    <style>
        body {
            margin: 0;
            padding: 0;
            background-color: #222;
            display: flex;
            justify-content: center;
            align-items: center;
            height: 100vh;
            font-family: Arial, sans-serif;
            overflow: hidden;
        }
        #gameCanvas {
            background-color: #4da6ff;
            border: 4px solid #fff;
            box-shadow: 0 0 20px rgba(0,0,0,0.5);
        }
    </style>
</head>
<body>

    <canvas id="gameCanvas" width="400" height="600"></canvas>

    <script>
        const canvas = document.getElementById("gameCanvas");
        const ctx = canvas.getContext("2d");

        // Game states
        let gameOver = false;

        // Game variables
        let frogY = 300;
        let frogX = 80;
        let frogSize = 30;
        let velocity = 0;
        let gravity = 0.4;
        let jumpStrength = -7;
        let score = 0;

        // Lily pad variables
        let lilies = [];
        let lilySpeed = 2;
        let spawnTimer = 0;

        function resetGame() {
            frogY = 300;
            velocity = 0;
            lilies = [];
            score = 0;
            spawnTimer = 0;
            gameOver = false;
        }

        function spawnLily() {
            let randomY = Math.floor(Math.random() * 200) + 350;
            let isPurple = Math.random() < 0.2; 
            
            lilies.push({
                x: canvas.width,
                y: randomY,
                width: 60,
                height: 15,
                purple: isPurple
            });
        }

        function checkCollision(frogX, frogY, frogSize, lily) {
            // Simple bounding box collision detection
            return frogX < lily.x + lily.width &&
                   frogX + frogSize > lily.x &&
                   frogY < lily.y + lily.height &&
                   frogY + frogSize > lily.y;
        }

        function update() {
            if (gameOver) return;

            // Apply physics to frog
            velocity += gravity;
            frogY += velocity;

            // LOSE CONDITION 1: Falling into the deep water
            if (frogY + frogSize > canvas.height) {
                gameOver = true;
            }

            // Spawn lilies
            spawnTimer++;
            if (spawnTimer > 100) {
                spawnLily();
                spawnTimer = 0;
            }

            // Move lilies
            for (let i = lilies.length - 1; i >= 0; i--) {
                lilies[i].x -= lilySpeed;

                // LOSE CONDITION 2: Hitting a purple carnivorous lily pad
                if (lilies[i].purple && checkCollision(frogX, frogY, frogSize, lilies[i])) {
                    gameOver = true;
                }

                // Remove old lilies and earn points
                if (lilies[i].x + lilies[i].width < 0) {
                    lilies.splice(i, 1);
                    score++;
                }
            }
        }

        function draw() {
            // Clear canvas with water color
            ctx.fillStyle = "#4da6ff";
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            // Draw frog (green square)
            ctx.fillStyle = "#2ecc71";
            ctx.fillRect(frogX, frogY, frogSize, frogSize);

            // Draw lily pads
            for (let lily of lilies) {
                ctx.fillStyle = lily.purple ? "#9b59b6" : "#3498db";
                ctx.fillRect(lily.x, lily.y, lily.width, lily.height);
            }

            // Draw score
            ctx.fillStyle = "#fff";
            ctx.font = "24px Arial";
            ctx.fillText("Score: " + score, 20, 40);

            // Draw Game Over Screen
            if (gameOver) {
                ctx.fillStyle = "rgba(0, 0, 0, 0.6)";
                ctx.fillRect(0, 0, canvas.width, canvas.height);

                ctx.fillStyle = "#fff";
                ctx.font = "40px Arial";
                ctx.textAlign = "center";
                ctx.fillText("GAME OVER", canvas.width / 2, canvas.height / 2 - 20);
                
                ctx.font = "20px Arial";
                ctx.fillText("Press Space to Restart", canvas.width / 2, canvas.height / 2 + 20);
                ctx.textAlign = "start"; // Reset text alignment
            }
        }

        function gameLoop() {
            update();
            draw();
            requestAnimationFrame(gameLoop);
        }

        // Listen for keyboard input
        window.addEventListener("keydown", function(e) {
            if (e.code === "Space") {
                if (gameOver) {
                    resetGame();
                } else {
                    velocity = jumpStrength;
                }
            }
        });

        // Start the game loop
        gameLoop();
    </script>
</body>
</html>
